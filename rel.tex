\begin{prop}
	{(Diaconescu 定理)}
	设 $\mathsf S$ 为\topos{}, $\mathcal C$ 为 $\mathsf S$ 的内蕴范畴, $p\colon \mathsf E\to\mathsf S$ 为相对\topos{}. 那么有范畴等价
	\[
	\operatorname{Hom}_{\Top/\mathsf S}
	(\mathsf E,\mathsf {Fun}(\mathcal C,\mathsf S))
	\simeq \mathsf {Tors}(\mathcal C,\mathsf E).
	\]
\end{prop}
